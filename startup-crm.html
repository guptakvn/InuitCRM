⚙ First-time setup

create table leads (id uuid primary key default gen_random_uuid(), user_id uuid references auth.users not null, name text, contact text, value numeric default 0, stage text default 'New lead', notes text, created_at timestamp default now());
create table contacts (id uuid primary key default gen_random_uuid(), user_id uuid references auth.users not null, name text, company text, phone text, email text, gst text, city text, created_at timestamp default now());
create table invoices (id uuid primary key default gen_random_uuid(), user_id uuid references auth.users not null, number text, client text, date text, subtotal numeric default 0, gst_rate numeric default 18, gst_amt numeric default 0, total numeric default 0, status text default 'Draft', created_at timestamp default now());
create table expenses (id uuid primary key default gen_random_uuid(), user_id uuid references auth.users not null, description text, amount numeric default 0, category text, date text, created_at timestamp default now());
alter table leads enable row level security;
